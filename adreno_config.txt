# ========================================
# Adreno GPU Driver Module Configuration
# ========================================
# Edit these values to customize module behavior
# y = enabled, n = disabled

# Public Libraries Patching
# Patches /vendor/etc/public.libraries*.txt files to expose gpu++.so
# Required for zuras banch++ to work properly if available 
# Recommended: n
PLT=n

# QGL Configuration
# Installs qgl_config.txt to /data/vendor/gpu/
# Enables additional Vulkan extensions and features
# Recommended: n (only for A8xx GPUs)
QGL=n

# Per-App QGL Configuration
# Only applies when QGL=y. Ignored entirely when QGL=n.
#
# y = per-app mode: QGLTrigger APK (AccessibilityService) monitors every app
#     switch and applies the matching qgl_config.txt.<package> file atomically.
#     The APK is auto-installed and root-granted during module flash.
#
# n = static mode: a single qgl_config.txt is written once at boot from the
#     bundled qgl_config.txt file (retry+verify logic, old code method).
#     No APK. No per-app switching. One config for everything.
#
# Recommended: y (default)
QGL_PERAPP=y

# QGL for System Apps
# Only applies when QGL=y and QGL_PERAPP=y.
#
# n = (default) skip system apps — the AccessibilityService ignores app switches
#     from system apps (SystemUI, launcher, settings, etc.). This prevents black
#     screens caused by applying QGL to apps that already have an active Vulkan
#     device at switch time.
#
# y = apply QGL to ALL apps including system apps. Use with caution — may cause
#     black screen on some devices if QGL is applied while the app's Vulkan
#     device is already active.
#
# Recommended: n
QGL_SYSTEM_APPS=n

# ========================================
# Advanced Options
# ========================================

# ARM64 Optimization
# Automatically removes 32-bit vendor libs on ARM64-only devices
# Saves storage space and reduces conflicts
# Recommended: n (set to y if your device is ARM64-only with no 32-bit apps)
ARM64_OPT=n

# Verbose Logging
# Enables detailed logging during installation
# Useful for troubleshooting installation issues
# Recommended: n (set to y if having issues)
VERBOSE=n

# ========================================
# Rendering Options
# ========================================

# Render Mode
# Controls the graphics rendering backend used by Android
# Options:
#   normal      - Default Android rendering (balanced, most compatible)
#   skiavk      - Skia with Vulkan backend (better performance on modern GPUs)
#   skiagl      - Skia with OpenGL backend (better compatibility with older apps/games)
#
# Details:
#   normal      - Standard Android renderer, no modifications
#   skiavk      - Sets debug.hwui.renderer=skiavk for improved GPU performance
#   skiagl      - Sets debug.hwui.renderer=skiagl for OpenGL compatibility
#
# Recommended: normal (change only if you experience rendering issues or want better performance)
# Advanced users: Try skiavk for performance, skiagl for compatibility
RENDER_MODE=normal

# ========================================
# SurfaceFlinger Backend Override
# ========================================

# Force Skiavkthreaded Backend
# Forces debug.renderengine.backend=skiavkthreaded even on Android < 14 (API 34).
# On Android 14+, this is set automatically when RENDER_MODE=skiavk.
# On Android < 14: SurfaceFlinger does not implement SkiaVkRenderEngine — strict OEM
# ROMs (MIUI/HyperOS, OneUI) will BOOTLOOP if an unknown backend string is passed.
# Only enable if you know your device's SurfaceFlinger supports skiavkthreaded.
# y = force skiavkthreaded backend regardless of Android version
# n = auto-detect (skiavkthreaded on API >= 34, skiaglthreaded otherwise)
FORCE_SKIAVKTHREADED_BACKEND=n
